: \mathit{field}    ::= T \ f; \\
: \mathit{pred}     ::= \code{predicate} \ p(\overline{T \ x}) = \tphi \\
: T                 ::= \code{int} | C | \top \\
: \mathit{method}   ::= T \ m (\overline{T \ x}) \ contract \ \{ s \} \\
: \mathit{contract} ::= \code{requires} \ \tphi \ \code{ensures} \ \tphi \\
: \oplus            ::= + | - | \ast | \backslash \\
: \odot             ::= \neq | = | < | > | \leq | \geq \\
: s                 ::= \code{skip}
                         | s_1 \ ; \ s_2
                         | T\ x | x := e
                         | \code{if} \ (e) \ \{s_1\} \ \code{else} \ \{s_2\}
                         | \code{while} \ (e) \ \code{inv} \ \tphi \ \{ s \}
                         | x.f := y
                         | x := \code{new} \ C
                         | y := z.m(\overline{x})
                         % | y := z.m_C(\overline{x}) % TODO: not sure what this is for
                         | \code{assert} \ \phi
                         % TODO: should the following be removed? they aren't in ap-main
                         % | \code{release} \ \phi
                         % | \code{hold} \ \phi \ \{ s \}
                         | \code{fold} \ p(\overline{e})
                         | \code{unfold} \ p(\overline{e})
                         \\
: e                 ::= v | x | e \oplus e | e \odot e | e.f \\
: x                 ::= \code{result} | id | \code{old}(id) | \code{this} \\
: v                 ::= n | o | \code{null} | \code{true} | \code{false} \\

                        \\
