    {~}
    {\evale {v} {v}}

    {~}
    {\evale {x} {\rho(x)}}

        \evale {e_1} {v_1}\\

    }
    {\evale {e_1 \oplus e_2} {v_1 \oplus v_2}}
    
    
    \inferrule* [right=EvComp]
    {
        \evale {e_1} {v_1}\\
        \evale {e_2} {v_2}
    }
    {\evale {e_1 \odot e_2} {v_1 \odot v_2}}
